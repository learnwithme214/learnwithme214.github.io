function filterPapers() { const dept = document.getElementById("department").value; const year = document.getElementById("year").value; const sem = document.getElementById("semester").value; const results = document.getElementById("results"); results.innerHTML = ""; if (!dept || !year) { results.innerHTML = "

Select department and year

"; return; } const base = "https://learnwithme214.github.io/main.html/papers"; let baseName = ""; if (dept === "btech") { if (!sem) { results.innerHTML = "

Select semester

"; return; } baseName = `${dept}_${sem}_${year}`; } else { baseName = `${dept}_${year}`; } let promises = []; for (let i = 1; i <= 5; i++) { let fileName = `${baseName}_${i}.pdf`; let fullPath = base + fileName; let p = fetch(fullPath).then(res => { if (res.ok) { return fullPath; } return null; }).catch(() => null); promises.push(p); } Promise.all(promises).then(values => { const validFiles = values.filter(v => v !== null); if (validFiles.length === 0) { results.innerHTML = "

No files found

"; return; } validFiles.forEach(fullPath => { const fileName = fullPath.split("/").pop(); const link = document.createElement("a"); link.href = fullPath; link.textContent = `📄 ${fileName}`; link.target = "_blank"; const download = document.createElement("a"); download.href = fullPath; download.textContent = "⬇️ Download"; download.className = "download"; download.download = fileName; results.appendChild(link); results.appendChild(download); }); }); }
